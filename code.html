import itertools

cities = int ( input (" Enter the number of cities : "))

data = {}
for i in range (1 , cities +1):
    for j in range (1 , cities +1):
        if i != j:
            city1 = input (f" Enter the name of city {i}: ")
            city2 = input (f" Enter the name of city {j}: ")
            distance = float ( input (f" Enter the distance between { city1 } and { city2 }: "))
            data [(i , j )] = distance
combinations = list ( itertools . permutations ( range (1 , cities +1)))
shortest_distance = float ( 'inf')
shortest_tour = None
for tour in combinations :
    distance = 0
    for i in range ( cities -1):
        distance += data [( tour [i], tour [i+1])]
    distance += data [( tour [-1], tour [0])]
    if distance < shortest_distance:
        shortest_distance = distance
        shortest_tour = tour

print (" Shortest tour :", end =" ")
for city in shortest_tour :
    print (f" City { city }", end =" -> ")
print (f" City { shortest_tour [0]}")
print (" Shortest distance :", shortest_distance )
